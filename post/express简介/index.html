安装

1

npm install express -S

基本使用

新建一个app.js，内容如下。之后可以用node运行这个文件，就可以建立一个本地服务器。

const express = require('express');// 引入express

const app = new express();;// 创建express对象

app.get('/', (resquest, response) => {
	response.send("hello,world!")
})

// 监听服务器
app.listen(3000,() => {
	console.log('express is running at port 3000')
})

其中 app.get() 方法即对应一个get请求，一个请求对应一个函数，post请求就对应 app.post()方法等等。

每个函数有两个参数

  1. 请求路径
  2. 回调函数，其中有两个参数，request和response，这其实是两个对象，里面有很多方法

例如 response.send() 就相当于原生的 response.write() 但功能更高级，比如不用再写响应头了

例如 request.query() 可以直接以对象形式获取查询参数

app.get('/index', (req, res) => {
    let ret = req.query   // 获取到一个对象
    res.send(ret.curPage);

    //可以在请求的时候传入查询参数：
    //http://localhost:3000/index?curPage=3&perPage=10
});

在express中导入某个文件时，如果导入的文件名是index那么可以省略这个文件名

获取请求数据：body-parser模块

express自带此模块，当然也可以用npm下载

1

npm i body-parser

此模块可以更简单的解析JSON格式的数据，使用方法如下

const bodyParser = require("body-parser");
const express = require('express');

const app = new express()

// 2、bodyParser功能添加到项目app中       注册到app对象下(把这个模块和app进行关联)
// parse application/x-www-form-urlencoded   针对普通页面提交功能
app.use(bodyParser.urlencoded({ extended: false }))  //false接收的值为字符串或者数组，true则为任意类型
// parse application/json    
app.use(bodyParser.json())   // 解析json格式

app.post('/register',(res,rep) => {
  let {username, email, password, repwd} = res.body;//使用body方法获得数据
  console.log(username, email, password, repwd)
  rep.send("post请求成功");// send别忘了
})
